macro(UNIT_TEST filename lib)
   add_executable(        ${filename}  ${filename}.cpp )
   target_link_libraries( ${filename}  ${lib})
   add_test(              ${filename}  ${CMAKE_CURRENT_BINARY_DIR}/${filename} )
endmacro(UNIT_TEST)
